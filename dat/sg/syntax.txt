START ->
	SCOPE_DECLARATION DECLARATIONS eof
	| eof

DECLARATIONS ->
	SCOPE_DECLARATION DECLARATIONS
	| eps
	
SCOPE_DECLARATION ->
	TOKEN TOKENS SCOPE

TOKEN_DECLARATION ->
	= TOKEN_DECLARATION_BODY

TOKEN_DECLARATION_BODY ->
	SYMBOL_EXPRESSION
	| ATTRIBUTE

SCOPES ->
	TOKEN TOKENS SCOPE SCOPES
	| eps

SCOPE ->
	TOKEN_DECLARATION DECLARATION_END
	| DECLARATION_END
	| CLASS_SCOPE
	| DOUBLE_COLON BASE_CLASS CLASS_SCOPE
	| CODE_BLOCK
	| DISPATCH_DECLARATION DECLARATION_END
	| COLON PARAMETER PARAMETER_BODY METHOD_BODY
	| OPENING_BRACKET SIGNATURE_BODY METHOD_BODY
	| RESEND_OPERATOR RESEND

CLASS_SCOPE ->
	{ SCOPES CLOSING_CURLY_BRACKET

PARAMETER_BODY ->
	: PARAMETER NEXT_PARAMETER
	| eps

NEXT_PARAMETER ->
	: PARAMETER NEXT_PARAMETER
	| eps
	
SIGNATURE_BODY ->
	TOKEN_PARAM PARAMETER NEXT_SIGNATURE

NEXT_SIGNATURE ->
	, TOKEN_PARAM PARAMETER NEXT_SIGNATURE
	| CLOSING_BRACKET

METHOD_BODY ->
	CODE_BLOCK
	| = RET_STATEMENT DECLARATION_END
	| => DISPATCH DECLARATION_END
	| <= RESEND

DISPATCH_DECLARATION ->
	=> DISPATCH 
	
CODE_BLOCK ->
	OPEN_SBRACKET STATEMENTS

DISPATCH ->
	OBJECT

RESEND ->
	MESSAGE RESEND_BODY

RESEND_BODY ->
	DECLARATION_END

STATEMENTS ->
	ROOT_EXPRESSION NEXT_STATEMENT
	| ^ RET_STATEMENT RET_STATEMENT_END
	| BLOCK_END

NEXT_STATEMENT ->
	STATEMENT_END STATEMENTS
	| BLOCK_END	

RET_STATEMENT ->
	EXPRESSION

SYMBOL_EXPRESSION ->
	EXPRESSION

ROOT_EXPRESSION ->
	identifier OPERATIONS ROOT_OPERATIONS
	| private OPERATIONS ROOT_OPERATIONS
	| reference OPERATIONS ROOT_OPERATIONS

SUB_EXPRESSION ->
	: PARAMETER CLOSURE_ARGS

CLOSURE_ARGS ->
	: PARAMETER CLOSURE_ARGS
	| ) CLOSURE_ARGS_BODY

CLOSURE_ARGS_BODY ->
	CODE_BLOCK
	| ( RET_STATEMENT )

EXPRESSION ->
        OBJECT OPERATIONS

OBJECT ->
	identifier OBJECT_BODY
	| private
	| reference OBJECT_BODY
	| integer
	| hex
	| long
	| real
	| literal
	| wide
	| CODE_BLOCK	
	| OPENING_BRACKET SUB_EXPRESSION

OBJECT_BODY ->
	:: OBJECT_EXT
	| eps

OBJECT_EXT ->
	NESTED_CLASS

ROOT_OPERATIONS ->
	:= ASSIGNING
	| L7_OPERATION
	| eps

OPERATIONS ->
	MESSAGE NEXT_MESSAGE
	| ( SUB_BODY
        | L6_OPERATION
	| eps

SUB_BODY ->
	MESSAGE_PARAMETER SUB_MESSAGE_PARAMETERS
	| IDLE_MESSAGE_PARAMETER

IDLE_MESSAGE_PARAMETER ->
	CLOSING_BRACKET

SUB_MESSAGE_PARAMETERS ->
	, MESSAGE_PARAMETER SUB_MESSAGE_PARAMETERS
	| CLOSING_BRACKET

MESSAGE_OPERATIONS ->
	MESSAGE NEXT_MESSAGE
	| eps

NEXT_MESSAGE ->
	; MESSAGE NEXT_MESSAGE
	| eps

MESSAGE ->
	identifier MESSAGE_BODY
	| private MESSAGE_BODY

MESSAGE_BODY ->
	: MESSAGE_PARAMETER
	| ( MESSAGE_LIST
	| eps

MESSAGE_LIST ->
	EXPRESSION MESSAGE_LIST_R

MESSAGE_LIST_R  ->
	, EXPRESSION NEXT_MESSAGE_LIST_ITEM
	| CLOSING_BRACKET

NEXT_MESSAGE_LIST_ITEM ->
	, EXPRESSION NEXT_MESSAGE_LIST_ITEM
	| CLOSING_BRACKET

ASSIGNING ->
	EXPRESSION

NESTED_CLASS ->
	OPENING_CURLY_BRACKET SCOPES CLOSING_CURLY_BRACKET

MESSAGE_PARAMETER ->
	OBJECT

L6_OPERATION ->
	< OBJECT TEMPLATE_BODY MESSAGE_OPERATIONS

L7_OPERATION ->
	? L7_OPERAND

L7_OPERAND ->
	CODE_BLOCK

TEMPLATE_BODY ->
	ANGLE_OPERATOR TEMPLATE_OPERAND
	| eps

TEMPLATE_OPERAND ->
	( EXPRESSION )
	| eps

DECLARATION_END ->
	TERMINATOR
	| DECLARATION_END_EXPECTED

TOKEN_BODY ->
	< FIRST_TOKEN_PARAM
	| eps

FIRST_TOKEN_PARAM ->
	TOKEN_PARAM TOKEN_PARAMS
	| SIZE_PARAM CLOSING_ANGLE_BRACKET

TOKENS ->
	TOKEN TOKENS
	| eps

TOKEN ->
	identifier TOKEN_BODY

TOKEN_PARAMS ->
	, TOKEN_PARAM TOKEN_PARAMS
	| SIZE_PARAM CLOSING_ANGLE_BRACKET
	| CLOSING_ANGLE_BRACKET

	identifier
	| reference
	| private

	identifier
	| reference 
	| private

	[

	}

	)

	>

BLOCK_END ->
	]
	| BLOCK_CLOSING_EXPECTED

	TERMINATOR
	| STATEMENT_END_EXPECTED

	TERMINATOR BLOCK_END
	| BLOCK_END

	.

	:

	::
