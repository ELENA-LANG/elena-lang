__define START          	1
__define eps            	2

__define TOKEN 			4112
__define GLOBAL_TOKEN 		4113
__define SCOPE_DECLARATION	4128
__define ATTRIBUTE_DECLARATION  4144
__define SIZE 			4160
__define BASE 			4176
__define EXPRESSION	       36912

__define eof            	8208
__define identifier 		8209
__define hex 			8210
__define reference		8211
__define global			8212

START ->
	DECLARATION DECLARATIONS eof
	| eof

ATTRIBUTE_DECLARATION ->
	identifier SIZE

DECLARATIONS ->
	DECLARATION DECLARATIONS
	| eps

DECLARATION ->
	TOKENS SCOPE_DECLARATION
	| #define ATTRIBUTE_DECLARATION

SCOPE_DECLARATIONS ->
	TOKENS SCOPE_DECLARATION SCOPE_DECLARATIONS
	| eps

SCOPE_DECLARATION ->
	CLASS_DECLARATION
	| SYMBOL_DECLARATION DECLARATION_END
	| DECLARATION_END

CLASS_DECLARATION ->
	{ SCOPE_DECLARATIONS }

SYMBOL_DECLARATION ->
	= EXPRESSION

EXPRESSION ->
	TOKEN
	| GLOBAL_TOKEN
	
TOKENS ->
	TOKEN NEXT_TOKEN

NEXT_TOKEN ->
	TOKENS
	| eps

TOKEN ->
	identifier
	| reference

GLOBAL_TOKEN ->
	global

SIZE ->
	hex
	| integer

DYNAMIC_SIZES ->
	DYNAMIC_SIZE DYNAMIC_SIZES
	| eps
	
DYNAMIC_SIZE ->
	[]

COLON ->
	:

DECLARATION_END ->
	;
