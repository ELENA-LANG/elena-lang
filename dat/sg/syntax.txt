START ->
	DECLARATION DECLARATIONS eof
	| eof

DECLARATION ->
        TOKEN NEXT_DECLARATION_ITEM

DECLARATIONS ->
	DECLARATION DECLARATIONS
	| eps

NEXT_DECLARATION_ITEM ->
	TOKEN NEXT_DECLARATION_ITEM
	| SCOPE_DECLARATION

SCOPE_DECLARATION ->
        CLASS_DECLARATION
	| SYMBOL_DECLARATION DECLARATION_END

CLASS_DECLARATION ->
	{ SCOPE_DECLARATIONS }

SCOPE_DECLARATIONS ->
	TOKEN NEXT_DECLARATION_ITEM SCOPE_DECLARATIONS
	| eps

SYMBOL_DECLARATION ->
	= EXPRESSION

EXPRESSION ->
        TOKEN TI_OPERATIONS

TOKEN ->
	identifier
	| reference

TI_OPERATIONS ->
	L2_OPERATION
	| OPERATIONS

OPERATIONS ->
	eps

L2_OPERATION ->
	. L2_OPERATOR_BODY

L2_OPERATOR_BODY ->
	MESSAGE MESSAGE_BODY

MESSAGE_BODY ->
	MESSAGE_OP MESSAGE_PARAMETERS )

MESSAGE_OP ->
	( 

MESSAGE_PARAMETERS ->
	eps

MESSAGE ->
	identifier	

DECLARATION_END ->
	;
