__define START          	1
__define eps            	2

__define TOKEN 			4112
__define SCOPE_DECLARATION	4128
__define EXPRESSION	       36912

__define eof            	8208
__define identifier 		8209

START ->
	DECLARATION DECLARATIONS eof
	| eof

DECLARATIONS ->
	DECLARATION DECLARATIONS
	| eps

DECLARATION ->
	TOKENS SCOPE_DECLARATION

SCOPE_DECLARATION ->
	SYMBOL_DECLARATION DECLARATION_END

SYMBOL_DECLARATION ->
	= EXPRESSION

EXPRESSION ->
	TOKEN
	
TOKENS ->
	TOKEN

TOKEN ->
	identifier

REFEREMCE_TOKEN ->
	reference

GLOBAL_TOKEN ->
	global

LITERAL ->
	literal
	| integer
	| hex
	| long
	| real
	| character
	| wide
	| constant

DYNAMIC_SIZES ->
	DYNAMIC_SIZE DYNAMIC_SIZES
	| eps
	
DYNAMIC_SIZE ->
	[]

DECLARATION_END ->
	;
