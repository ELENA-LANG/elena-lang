__define START          	1
__define eps            	2

__define TOKEN 			4112
__define SCOPE_DECLARATION	4128
__define EXPRESSION		4144

__define eof            	8208
__define identifier 		8209

START ->
	DECLARATION DECLARATIONS eof
	| eof

DECLARATIONS ->
	DECLARATION DECLARATIONS
	| eps

DECLARATION ->
	TOKEN_DECLARATION SCOPE_DECLARATION

SCOPE_DECLARATION ->
	SYMBOL_DECLARATION DECLARATION_END

SYMBOL_DECLARATION ->
	= EXPRESSION

EXPRESSION ->
	TOKEN_EXPRESSION 

TOKEN_EXPRESSION ->
	TOKEN
	
TOKEN_DECLARATIONS ->
	< TEMPLATE_ARGS > LAST_TOKEN_DECLARATION
	| DYNAMIC_SIZE DYNAMIC_SIZES TOKEN
	| TOKEN TOKEN_DECLARATIONS
	| eps

LAST_TOKEN_DECLARATION ->
	TOKEN
	| DYNAMIC_SIZE DYNAMIC_SIZES TOKEN

TOKEN_DECLARATION ->
	TOKEN TOKEN_DECLARATIONS

TEMPLATE_ARGS ->
	TOKEN_DECLARATION

TOKENS ->
	, TOKEN TOKENS
	| eps

TOKEN ->
	identifier
	| reference
	| global

LITERAL ->
	literal
	| integer
	| hex
	| long
	| real
	| character
	| wide
	| constant

DYNAMIC_SIZES ->
	DYNAMIC_SIZE DYNAMIC_SIZES
	| eps
	
DYNAMIC_SIZE ->
	[]

DECLARATION_END ->
	;
