__define START          	1
__define eps            	2

__define TOKEN 			4112
__define GLOBAL_TOKEN 		4113
__define SCOPE_DECLARATION	4128
__define ATTRIBUTE_DECLARATION  4144
__define SIZE 			4160
__define BASE 			4176
__define BLOCK 			4192
__define DISPATCH		4208
__define EXPRESSION	       36912

__define eof            	8208
__define identifier 		8209
__define hex 			8210
__define reference		8211
__define global			8212

START ->
	DECLARATION DECLARATIONS eof
	| eof

ATTRIBUTE_DECLARATION ->
	identifier SIZE

DECLARATIONS ->
	DECLARATION DECLARATIONS
	| eps

DECLARATION ->
	TOKENS SCOPE_DECLARATION
	| #define ATTRIBUTE_DECLARATION

SCOPE_DECLARATIONS ->
	TOKENS SCOPE_DECLARATION SCOPE_DECLARATIONS
	| eps

SCOPE_DECLARATION ->
	CLASS_DECLARATION
	| COLON BASE BASE_DECLARATION
	| SYMBOL_DECLARATION DECLARATION_END
	| METHOD_DECLARATION
	| DECLARATION_END

BASE_DECLARATION ->
	CLASS_DECLARATION
	| DECLARATION_END

CLASS_DECLARATION ->
	{ SCOPE_DECLARATIONS }

METHOD_DECLARATION ->
	( PARAMETERS ) METHOD_BODY

METHOD_BODY ->
	BLOCK
	| => DISPATCH DECLARATION_END

DISPATCH ->
	EXPRESSION

SYMBOL_DECLARATION ->
	= EXPRESSION

BLOCK ->
	OPENING_CURLY_BRACKET STATEMENTS

STATEMENTS ->
	BLOCK_END

EXPRESSION ->
	TOKEN T_OPERATIONS
	| GLOBAL_TOKEN

T_OPERATIONS ->
	TOKEN
	| eps

PARAMETERS ->
	PARAMETER NEXT_PARAMETER	
	| eps

PARAMETER ->
	TOKENS

NEXT_PARAMETER ->
	, PARAMETER NEXT_PARAMETER
	| eps
	
TOKENS ->
	TOKEN NEXT_TOKEN

NEXT_TOKEN ->
	TOKENS
	| eps

TOKEN ->
	identifier
	| reference

GLOBAL_TOKEN ->
	global

SIZE ->
	hex
	| integer

BASE ->
	TOKEN
	| GLOBAL_TOKEN

DYNAMIC_SIZES ->
	DYNAMIC_SIZE DYNAMIC_SIZES
	| eps
	
DYNAMIC_SIZE ->
	[]

OPENING_CURLY_BRACKET ->
	{

COLON ->
	:

DECLARATION_END ->
	;

BLOCK_END ->
	}
