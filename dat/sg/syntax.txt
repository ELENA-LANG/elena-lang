	SCOPE_DECLARATION DECLARATIONS eof
	| eof

DECLARATIONS ->
	SCOPE_DECLARATION DECLARATIONS
	| eps
	
SCOPES ->
	TOKEN TOKENS SCOPE SCOPES
	| eps

TOKENS ->
	TOKEN TOKENS
	| eps

TOKEN_PARAMS ->
	, TOKEN_PARAM TOKEN_PARAMS
	| )

SCOPE_DECLARATION ->
	TOKEN TOKENS SCOPE

SCOPE ->
	= TOKEN_DECLARATION DECLARATION_END
	| { SCOPES CLOSING_CURLY_BRACKET
	| => DISPATCH DECLARATION_END
	| CODE_BLOCK
	| : PARAMETER PARAMETER_BODY METHOD_BODY
	| ( BRACKET_BODY 
	| DECLARATION_END
	| :: BASE_CLASS SCOPE_BODY

SCOPE_BODY ->
	CLASS_BODY

CLASS_BODY ->
	{ SCOPES CLOSING_CURLY_BRACKET

BRACKET_BODY ->
	SIZE_PARAM CLOSING_BRACKET DECLARATION_END
	| SIGNATURE_BODY METHOD_BODY

TOKEN_DECLARATION ->
	SYMBOL_EXPRESSION
	| ATTRIBUTE

PARAMETER_BODY ->
	: PARAMETER NEXT_PARAMETER
	| SUBJECT_ARG : PARAMETER NEXT_SUBJECT_PARAMETER
	| eps
	        
NEXT_SUBJECT_PARAMETER ->
	SUBJECT_ARG : PARAMETER NEXT_SUBJECT_PARAMETER
	| eps

NEXT_PARAMETER ->
	: PARAMETER NEXT_PARAMETER
	| eps

SIGNATURE_BODY ->
	TOKEN_PARAM PARAMETER NEXT_SIGNATURE

NEXT_SIGNATURE ->
	, TOKEN_PARAM PARAMETER NEXT_SIGNATURE
	| CLOSING_BRACKET

METHOD_BODY ->
	CODE_BLOCK
	| = RET_STATEMENT DECLARATION_END
	| => DISPATCH DECLARATION_END

DISPATCH ->
	OBJECT

SYMBOL_EXPRESSION ->
	EXPRESSION

NESTED_CLASS ->
	{ SCOPES CLOSING_CURLY_BRACKET

CODE_BLOCK ->
	[ STATEMENTS

STATEMENTS ->
	EXPRESSION NEXT_STATEMENT
	| BLOCK_END

NEXT_STATEMENT ->
	STATEMENT_END STATEMENTS
	| BLOCK_END	

RET_STATEMENT ->
	EXPRESSION

EXTENSION ->
	~ OBJECT

EXPRESSION ->
	OBJECT OPERATIONS

MESSAGE_PARAMETER ->
	OBJECT

OBJECT ->
	identifier OBJECT_BODY
	| private
	| reference
	| % MESSAGE_REFERENCE
	| CODE_BLOCK	
	| NESTED_CLASS

OBJECT_BODY ->
	:: OBJECT_EXT
	| eps

OBJECT_EXT ->
	NESTED_CLASS

OPERATIONS ->
	MESSAGE NEXT_MESSAGE
	| EXTENSION MESSAGE NEXT_MESSAGE
	| eps

MESSAGE ->
	identifier MESSAGE_BODY

MESSAGE_BODY ->
	SUBJECT_ARG AFTER_FIRST_SUBJECT
	| : MESSAGE_PARAMETER NEXT_MESSAGE_PARAMETER
	| ( MESSAGE_LIST
	| eps

AFTER_FIRST_SUBJECT ->
	: MESSAGE_PARAMETER NEXT_MESSAGE_PARAMETER
	| eps

NEXT_MESSAGE_PARAMETER ->
	: MESSAGE_PARAMETER MESSAGE_PARAMETERS
	| SUBJECT_ARG : MESSAGE_PARAMETER SUBJ_MESSAGE_PARAMETERS
	| eps

MESSAGE_PARAMETERS ->
	: MESSAGE_PARAMETER MESSAGE_PARAMETERS
	| eps

SUBJ_MESSAGE_PARAMETERS ->
	SUBJECT_ARG : MESSAGE_PARAMETER SUBJ_MESSAGE_PARAMETERS
	| eps

MESSAGE_LIST ->
	EXPRESSION NEXT_MESSAGE_LIST_ITEM

NEXT_MESSAGE_LIST_ITEM ->
	, EXPRESSION NEXT_MESSAGE_LIST_ITEM
	| CLOSING_BRACKET
	      
NEXT_MESSAGE ->
	; MESSAGE NEXT_MESSAGE
	| eps
	
DECLARATION_END ->
	TERMINATOR
	| DECLARATION_END_EXPECTED

TOKEN ->
	identifier

ATTRIBUTE ->
	attribute

SUBJECT_ARG ->
	identifier

PARAMETER ->
	identifier

TOKEN_PARAM ->
	identifier
	| reference

MESSAGE_REFERENCE ->
	identifier
	| literal

SIZE_PARAM ->
	integer
	
BASE_CLASS ->
	identifier TOKEN_BODY
	| reference

	( TOKEN_PARAM TOKEN_PARAMS
	| eps	

TERMINATOR ->
	.

CLOSING_BRACKET ->
	)

	TERMINATOR
	| STATEMENT_END_EXPECTED

BLOCK_END ->
	]
	| BLOCK_CLOSING_EXPECTED

	}
